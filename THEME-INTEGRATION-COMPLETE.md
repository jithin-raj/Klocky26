# 🎨 Complete Theme Integration Guide

## Overview
Your API stores **only ONE color** per organization. The frontend automatically generates all other theme colors from this single color.

---

## 📦 What Your Backend Stores

### Database Schema
```sql
CREATE TABLE organizations (
  id VARCHAR(255) PRIMARY KEY,
  name VARCHAR(255) NOT NULL,
  brand_color VARCHAR(7) DEFAULT '#0d9488',  -- Just one color!
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
```

### Example Data
```sql
INSERT INTO organizations VALUES 
  ('acme', 'Acme Corp', '#6366f1'),
  ('globex', 'Globex Inc', '#d97706');
```

## 🔄 Integration Flow

### 1. After Login (Apply Org Theme)

```typescript
// auth.service.ts
import { OrgThemeService } from './core/services/org-theme.service';

@Injectable({ providedIn: 'root' })
export class AuthService {
  private orgThemeService = inject(OrgThemeService);

  async login(email: string, password: string): Promise<void> {
    const response = await this.api.post('/auth/login', { email, password });
    
    // API returns: { organization: { brandColor: "#6366f1" } }
    if (response.organization?.brandColor) {
      // Generate complete theme from single color
      const theme = this.orgThemeService.generateThemeFromColor(
        response.organization.brandColor
      );
      
      // Apply to entire application
      this.orgThemeService.apply(theme);
    }
  }
}
```

### 2. On App Initialization

```typescript
// shell.component.ts or app.component.ts
export class ShellComponent implements OnInit {
  private orgThemeService = inject(OrgThemeService);

  async ngOnInit(): Promise<void> {
    // Restore theme on page reload
    if (this.isLoggedIn()) {
      const org = await this.api.get('/organizations/me');
      
      if (org.brandColor) {
        const theme = this.orgThemeService.generateThemeFromColor(org.brandColor);
        this.orgThemeService.apply(theme);
      }
    }
  }
}
```

### 3. Save from Settings

```typescript
// Already implemented in org-profile.component.ts
async save(): Promise<void> {
  // Send only the accent color to API
  await this.api.put('/organizations/profile', {
    companyName: this.companyName,
    brandColor: this.accentColor  // Single color: "#6366f1"
  });
  
  // Apply theme immediately
  const theme = this.orgThemeService.generateThemeFromColor(this.accentColor);
  this.orgThemeService.apply(theme);
}
```

## 🎨 Example Backend Code

### Node.js/Express
```javascript
// GET /organizations/me
app.get('/organizations/me', authenticate, async (req, res) => {
  const org = await db.organizations.findOne({ 
    id: req.user.organizationId 
  });
  
  res.json({
    id: org.id,
    name: org.name,
    brandColor: org.brand_color || '#0d9488'  // Default teal
  });
});

// PUT /organizations/:id/profile
app.put('/organizations/:id/profile', authenticate, async (req, res) => {
  const { brandColor } = req.body;
  
  // Validate hex color
  if (!/^#[0-9A-F]{6}$/i.test(brandColor)) {
    return res.status(400).json({ error: 'Invalid color format' });
  }
  
  await db.organizations.update(
    { id: req.params.id },
    { brand_color: brandColor }
  );
  
  res.json({ success: true });
});
```
